## Install

```bash
npm install --save @mana/base
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from '@mana/base'
import '@mana/base/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
